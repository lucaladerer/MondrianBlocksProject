Blocks Klasse Ideen:

bool insert(int ankerPos[2], char size[2])
{
    if(game.Checkfit(size)){
        short xpos = ankerPos[0];
        short ypos = ankerPos[1];
        short maxX = size[0];
        short maxY = size[1];
        for(short x = xpos; x <= maxX; x++){
            for(short y = ypos; y <= maxY; y++){
                fieldset.template(x, y);
        }
    }
        return true;
    }
    else return false;
}
